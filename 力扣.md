2. 两数相加

```
输入：l1 = [2,4,3], l2 = [5,6,4]
输出：[7,0,8]
解释：342 + 465 = 807.
链表 7-> 0 -> 8
```

```
package main

import (
	"fmt"
	"strconv"
)

type ListNode struct {
	Val  int
	Next *ListNode
}

func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
	var l1_List = []int{}
	var l2_List = []int{}
	for {
		l1_List = append(l1_List, l1.Val)
		if l1.Next == nil {
			break
		}
		l1 = l1.Next
	}
	for {
		l2_List = append(l2_List, l2.Val)
		if l2.Next == nil {
			break
		}
		l2 = l2.Next
	}

	var aa = ""
	var bb = ""

	for k := range l1_List {
		aa += strconv.Itoa(l1_List[len(l1_List)-1-k])
	}

	for k := range l2_List {
		bb += strconv.Itoa(l2_List[len(l2_List)-1-k])
	}
	tmp1, _ := strconv.Atoi(aa)
	tmp2, _ := strconv.Atoi(bb)
	tmp3 := tmp1 + tmp2
	z := strconv.Itoa(tmp3)
	var n *ListNode
	for _, v := range z {
		tt := string(v)
		ttt, _ := strconv.Atoi(tt)
		tmp_node := &ListNode{
			Val:  ttt,
			Next: nil,
		}
		tmp_node.Next = n
		n = tmp_node
	}
	return n
}

func main() {
	// 342 + 465 = 807
	var a = &ListNode{
		Val: 2,
		Next: &ListNode{
			Val: 4,
			Next: &ListNode{
				Val:  3,
				Next: nil,
			},
		},
	}
	var b = &ListNode{
		Val: 5,
		Next: &ListNode{
			Val: 6,
			Next: &ListNode{
				Val:  4,
				Next: nil,
			},
		},
	}
	c := addTwoNumbers(a, b)
	fmt.Println(c)
}

```
package main

import (
	"fmt"
)

type ListNode struct {
	Val  int
	Next *ListNode
}

func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
	var n = &ListNode{}
	t := 0
	for {
		if l1 == nil && l2 == nil && t == 0 {
			break
		}
		if l1 != nil {
			t += l1.Val
			l1 = l1.Next
		}
		if l2 != nil {
			t += l2.Val
			l2 = l2.Next
		}
		n.Next = &ListNode{
			Val: t % 10,
		}
		n = n.Next
		t = t / 10
	}
	return n
}

func main() {
	var a = &ListNode{
		Val: 2,
		Next: &ListNode{
			Val: 4,
			Next: &ListNode{
				Val:  3,
				Next: nil,
			},
		},
	}
	var b = &ListNode{
		Val: 5,
		Next: &ListNode{
			Val: 6,
			Next: &ListNode{
				Val:  4,
				Next: nil,
			},
		},
	}
	c := addTwoNumbers(a, b)
	fmt.Println("c = ", c)
}
